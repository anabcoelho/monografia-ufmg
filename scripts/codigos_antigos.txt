# Valores ajustados
fitted_vals <- fitted(modelo_logit)

# 1) Resíduos de deviance
res_dev <- residuals(modelo_logit, type = "deviance")

plot(
  fitted_vals, res_dev,
  pch = 19, col = "gray50",
  xlab = "Probabilidade estimada",
  ylab = "Resíduo de deviance",
  main = "Deviance vs Ajustados"
)
abline(h = 0, lty = 2)
grid()

# 2) Resíduos de Pearson
res_pearson <- residuals(modelo_logit, type = "pearson")

plot(
  fitted_vals, res_pearson,
  pch = 19, col = "gray50",
  xlab = "Probabilidade estimada",
  ylab = "Resíduo de Pearson",
  main = "Pearson vs Ajustados"
)
abline(h = 0, lty = 2)
grid()

# 3) QQ-plot dos resíduos de deviance
qqnorm(
  res_dev,
  main = "QQ-plot (Resíduos de deviance)",
  pch = 19, col = "gray50"
)
qqline(res_dev, col = "red", lwd = 2)
grid()

# 4) Alavancagem (hat values)
hat_vals <- hatvalues(modelo_logit)

plot(
  hat_vals,
  type = "h",
  lwd = 2,
  col = "gray50",
  xlab = "Observações",
  ylab = "Alavancagem",
  main = "Valores de alavancagem"
)
abline(h = 2 * mean(hat_vals), col = "red", lty = 2)
grid()
